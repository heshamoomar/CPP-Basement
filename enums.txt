#include <iostream>
using namespace std;

enum enScreenColor { Red = 1, Blue = 2, Green = 3, Yellow = 4 };

int main()
{
    cout << "****************************\n";
    cout << "Please Chose the number of your color?\n";
    cout << "(1) Red\n";
    cout << "(2) Blue\n";
    cout << "(3) Green\n";
    cout << "(4) Yellow\n";
    cout << "****************************\n\n";
    cout << "Your Choice? ";

    int c;
    enScreenColor Color;

    cin >> c;
    Color = (enScreenColor)c;

    switch (Color) {

    case Red:

        system("color 4F");
        break;

    case Blue:

        system("color 1F");
        break;

    case Green:

        system("color 2F");
        break;

    case Yellow:

        system("color 6F");
        break;

    default:
        system("color 0F");
    }

    return 0;
}
